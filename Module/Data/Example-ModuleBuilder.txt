param(
    [ValidateSet('Manifest', 'Build', 'Publish')]
    [string] $ConfigurationGateMode = 'Build',

    [bool] $SignModule = $false,

    [string] $CertificateThumbprint
)

Import-Module PSPublishModule -Force -ErrorAction Stop

Build-Module -ModuleName '$ModuleName' {
    New-ConfigurationManifest `
        -ModuleVersion '1.0.0' `
        -CompatiblePSEditions 'Desktop', 'Core' `
        -Guid '$Guid' `
        -Author 'Author' `
        -CompanyName 'CompanyName' `
        -Copyright "(c) 2011 - $((Get-Date).Year) Author @ CompanyName. All rights reserved." `
        -Description 'Simple project $ModuleName' `
        -PowerShellVersion '5.1' `
        -Tags 'Windows', 'MacOS', 'Linux'

    # Example dependency declarations:
    # New-ConfigurationModule -Type RequiredModule -Name 'PSSharedGoods' -Guid 'Auto' -Version 'Latest'
    # New-ConfigurationModule -Type ApprovedModule -Name 'PSSharedGoods', 'PSWriteColor'
    # New-ConfigurationModuleSkip -IgnoreFunctionName 'Invoke-Formatter', 'Find-Module'

    $profile = @{
        SignModule            = $SignModule
        ImportRequiredModules = $true
    }
    if (-not [string]::IsNullOrWhiteSpace($CertificateThumbprint)) {
        $profile.CertificateThumbprint = $CertificateThumbprint
    }

    New-ConfigurationModuleBuildProfile @profile

    # Optional artefacts:
    # New-ConfigurationArtefact -Type Unpacked -Enable -Path 'Artefacts\Unpacked\<TagModuleVersionWithPreRelease>'
    # New-ConfigurationArtefact -Type Packed -Enable -Path 'Artefacts\Packed' -IncludeTagName

    # Optional publish targets. API key files are read only by publish-gated runs.
    # New-ConfigurationPublish -Type PowerShellGallery -FilePath 'C:\Support\Important\PowerShellGalleryAPI.txt' -Enabled:$false
    # New-ConfigurationPublish -Type GitHub -FilePath 'C:\Support\Important\GitHubAPI.txt' -UserName 'CompanyName' -Enabled:$false

    New-ConfigurationGate -Mode $ConfigurationGateMode
} -ExitCode
