$Public = @(Get-ChildItem -LiteralPath (Join-Path $PSScriptRoot 'Public') -Filter '*.ps1' -Recurse -ErrorAction SilentlyContinue)
$Private = @(Get-ChildItem -LiteralPath (Join-Path $PSScriptRoot 'Private') -Filter '*.ps1' -Recurse -ErrorAction SilentlyContinue)
$Classes = @(Get-ChildItem -LiteralPath (Join-Path $PSScriptRoot 'Classes') -Filter '*.ps1' -Recurse -ErrorAction SilentlyContinue)
$Enums = @(Get-ChildItem -LiteralPath (Join-Path $PSScriptRoot 'Enums') -Filter '*.ps1' -Recurse -ErrorAction SilentlyContinue)

foreach ($Import in @($Classes + $Enums + $Private + $Public)) {
    try {
        . $Import.FullName
    } catch {
        throw "Failed to import '$($Import.FullName)': $($_.Exception.Message)"
    }
}

$ExportedFunctions = @($Public | ForEach-Object { $_.BaseName })
Export-ModuleMember -Function $ExportedFunctions
